### Come sono implementati

**1. Dark Grain** → `frag1_DarkGrain`
- base `vec3(0.04)` = #0a0a0a
- grana con `fract(sin(dot()))` classica
- vignette con `pow(length,1.8)`

**2. White Paper** → `frag2_WhitePaper`
- stessa formula, colore invertito `0.98`
- vignette al 10% per non essere piatto

**3. Grid Dark** → `frag3_GridDark`
- niente ripple, solo `fract(uv*12.0)`
- `smoothstep` crea linee da 1px
- opacità 7% su nero

**4. Grid White** → `frag4_GridWhite`
- identico, sfondo 0.99, linee nere al 6%

**5. Soft Mesh** → `frag5_SoftMesh`
- un solo `exp(-d*d*4.0)` per l'alone
- zero animazione, zero costo

## 1. Aurora Mesh
Nastri che respirano, zero geometria.

**aurora.frag**
```glsl
#ifdef GL_ES
precision mediump float;
#endif
uniform float time;
uniform vec2 resolution;

float noise(vec2 p){
    return fract(sin(dot(p, vec2(12.9898,78.233))) * 43758.5453);
}
float fbm(vec2 p){
    float v = 0.0;
    v += noise(p*1.0)*0.5;
    v += noise(p*2.0)*0.25;
    v += noise(p*4.0)*0.125;
    return v;
}

void main(){
    vec2 uv = gl_FragCoord.xy / resolution;
    uv.y *= resolution.y / resolution.x;

    vec3 col = vec3(0.04,0.04,0.06); // #0a0a0f

    for(int i=0; i<3; i++){
        float t = time*0.05 + float(i)*1.5;
        vec2 p = uv + vec2(sin(t), cos(t*0.7))*0.2;
        float n = fbm(p*2.0 + t);
        vec3 aurora = mix(vec3(0.0,0.7,0.8), vec3(0.5,0.2,0.9), float(i)/2.0);
        aurora = mix(aurora, vec3(0.9,0.2,0.6), n);
        col += aurora * 0.15 * smoothstep(0.3,0.7,n);
    }
    gl_FragColor = vec4(col,1.0);
}
```

uniform vec2 mouse;
uniform vec2 resolution;
uniform float time;
void main(){
    vec2 uv = gl_FragCoord.xy;

    vec2 grid = floor(uv/size);

include <glad/glad.h>
#include <GLFW/glfw3.h>
#include <iostream>
#include <cmath>

// ---------- SHADERS (identici al tuo React) ----------
const char* vertexShaderSrc = R"#(
#version 330 core
layout(location = 0) in vec2 position;
out vec2 vUv;
void main() {
    vUv = position * 0.5 + 0.5;
    gl_Position = vec4(position, 0.0, 1.0);
}
)#";

const char* fragmentShaderSrc = R"#(
#version 330 core
precision highp float;
uniform float iTime;
uniform vec2 iResolution;
uniform bool enableRainbow;
uniform vec3 gridColor;
uniform float rippleIntensity;
uniform float gridSize;
uniform float gridThickness;
uniform float fadeDistance;
uniform float vignetteStrength;
uniform float glowIntensity;
uniform float opacity;
uniform float gridRotation;
uniform bool mouseInteraction;
uniform vec2 mousePosition;
uniform float mouseInfluence;
uniform float mouseInteractionRadius;
in vec2 vUv;
out vec4 fragColor;

float pi = 3.141592;
mat2 rotate(float a){ float s=sin(a),c=cos(a); return mat2(c,-s,s,c); }

void main(){
    vec2 uv = vUv*2.0-1.0;
    uv.x *= iResolution.x/iResolution.y;
    if(gridRotation!=0.0) uv = rotate(gridRotation*pi/180.0)*uv;

    float dist = length(uv);
    float func = sin(pi*(iTime-dist));
    vec2 rippleUv = uv + uv*func*rippleIntensity;

    if(mouseInteraction && mouseInfluence>0.0){
        vec2 m = (mousePosition*2.0-1.0); m.x*=iResolution.x/iResolution.y;
        float md = length(uv-m);
        float infl = mouseInfluence*exp(-md*md/(mouseInteractionRadius*mouseInteractionRadius));
        float wave = sin(pi*(iTime*2.0-md*3.0))*infl;
        rippleUv += normalize(uv-m)*wave*rippleIntensity*0.3;
    }

    vec2 a = sin(gridSize*0.5*pi*rippleUv - pi/2.0);
    vec2 b = abs(a);
    vec2 sb = vec2(smoothstep(0.0,0.5,b.x), smoothstep(0.0,0.5,b.y));

    vec3 col = vec3(0.0);
    col += exp(-gridThickness*sb.x*(0.8+0.5*sin(pi*iTime)));
    col += exp(-gridThickness*sb.y);
    col += 0.5*exp(-(gridThickness/4.0)*sin(sb.x));
    col += 0.5*exp(-(gridThickness/3.0)*sb.y);
    col += glowIntensity*exp(-gridThickness*0.5*sb.x);
    col += glowIntensity*exp(-gridThickness*0.5*sb.y);

    float ddd = exp(-2.0*clamp(pow(dist,fadeDistance),0.0,1.0));
    float vign = 1.0 - pow(length(vUv-0.5)*2.0, vignetteStrength);
    vign = clamp(vign,0.0,1.0);

    vec3 t = enableRainbow? vec3(uv.x*0.5+0.5*sin(iTime),
                                  uv.y*0.5+0.5*cos(iTime),
                                  pow(cos(iTime),4.0))+0.5 : gridColor;

    float fade = ddd*vign;
    float alpha = length(col)*fade*opacity;
    fragColor = vec4(col*t*fade*opacity, alpha);
}
)#";

// ---------- PARAMS (match screenshot) ----------
struct Params {
    bool enableRainbow = false;
    float gridColor[3] = {0.416f, 0.231f, 1.0f}; // #6a3bff
    float rippleIntensity = 0.06f;
    float gridSize = 10.0f;
    float gridThickness = 22.0f;
    float fadeDistance = 1.8f;
    float vignetteStrength = 2.8f;
    float glowIntensity = 0.35f;
    float opacity = 1.0f;
    float gridRotation = 0.0f;
    bool mouseInteraction = true;
    float mouseInteractionRadius = 1.0f;
};

float mouseX=0.5f, mouseY=0.5f, targetX=0.5f, targetY=0.5f, mouseInfluence=0.0f;

void cursorPos(GLFWwindow* w,double x,double y){
    int W,H; glfwGetWindowSize(w,&W,&H);
    targetX = (float)x/W; targetY = 1.0f - (float)y/H;
}
void cursorEnter(GLFWwindow*,int e){ mouseInfluence = e? 1.0f : 0.0f; }

GLuint makeShader(GLenum t,const char* s){
    GLuint id=glCreateShader(t); glShaderSource(id,1,&s,nullptr); glCompileShader(id);
    int ok; glGetShaderiv(id,GL_COMPILE_STATUS,&ok);
    if(!ok){ char log[512]; glGetShaderInfoLog(id,512,nullptr,log); std::cerr<<log<<"\n"; }
    return id;
}

    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR,3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR,3);
    glfwWindowHint(GLFW_OPENGL_PROFILE,GLFW_OPENGL_CORE_PROFILE);
    GLFWwindow* win = glfwCreateWindow(1600,900,"RippleGrid C++",nullptr,nullptr);
    glfwMakeContextCurrent(win);
    gladLoadGLLoader((GLADloadproc)glfwGetProcAddress);
    glfwSetCursorPosCallback(win,cursorPos);
    glfwSetCursorEnterCallback(win,cursorEnter);

    // fullscreen triangle
    float verts[] = {-1,-1, 3,-1, -1,3};
    GLuint vao,vbo; glGenVertexArrays(1,&vao); glGenBuffers(1,&vbo);
    glBindVertexArray(vao); glBindBuffer(GL_ARRAY_BUFFER,vbo);
    glBufferData(GL_ARRAY_BUFFER,sizeof(verts),verts,GL_STATIC_DRAW);
    glVertexAttribPointer(0,2,GL_FLOAT,GL_FALSE,0,0); glEnableVertexAttribArray(0);

    GLuint prog = glCreateProgram();
    glAttachShader(prog,makeShader(GL_VERTEX_SHADER,vertexShaderSrc));
    glAttachShader(prog,makeShader(GL_FRAGMENT_SHADER,fragmentShaderSrc));
    glLinkProgram(prog); glUseProgram(prog);

    glEnable(GL_BLEND); glBlendFunc(GL_SRC_ALPHA,GL_ONE_MINUS_SRC_ALPHA);

    Params p;
    while(!glfwWindowShouldClose(win)){
        glfwPollEvents();
        mouseX += (targetX-mouseX)*0.1f;
        mouseY += (targetY-mouseY)*0.1f;

        int w,h; glfwGetFramebufferSize(win,&w,&h);
        glViewport(0,0,w,h);
        glClearColor(0.012f,0.0f,0.016f,1.0f); // sfondo viola scurissimo
        glClear(GL_COLOR_BUFFER_BIT);

        glUniform1f(glGetUniformLocation(prog,"iTime"), (float)glfwGetTime());
        glUniform2f(glGetUniformLocation(prog,"iResolution"), w, h);
        glUniform1i(glGetUniformLocation(prog,"enableRainbow"), p.enableRainbow);
        glUniform3f(glGetUniformLocation(prog,"gridColor"), p.gridColor[0],p.gridColor[1],p.gridColor[2]);
        glUniform1f(glGetUniformLocation(prog,"rippleIntensity"), p.rippleIntensity);
        glUniform1f(glGetUniformLocation(prog,"gridSize"), p.gridSize);

#include <SFML/Graphics.hpp>
#include <unordered_map>
#include <string>
#include <cmath>

// Configurazione identica allo screenshot
struct Config {
    float squareSize = 46.f;
    sf::Color background = sf::Color(14, 11, 18); // #0e0b12
    sf::Color borderColor = sf::Color(38, 34, 51); // #262233
    sf::Color hoverFill = sf::Color(26, 22, 35); // #1a1623
    float hoverMaxAlpha = 0.6f; // 60% come nella demo
    float fadeSpeed = 0.15f; // stesso lerp del tuo React
    float speed = 0.f; // 0 = griglia ferma
};

std::string key(int x, int y) {
    return std::to_string(x) + "," + std::to_string(y);
}

int main() {
    Config cfg;
    sf::RenderWindow window(sf::VideoMode(1280, 720), "ShapeGrid - Square", sf::Style::Resize);
    window.setFramerateLimit(60);

    sf::Vector2f gridOffset{0.f, 0.f};
    std::unordered_map<std::string, float> cellOpacities;
    sf::Vector2i hoveredCell = {-1, -1};

    while (window.isOpen()) {
        sf::Event ev;
        while (window.pollEvent(ev)) {
            if (ev.type == sf::Event::Closed) window.close();
            if (ev.type == sf::Event::Resized) {
                window.setView(sf::View(sf::FloatRect(0, 0, ev.size.width, ev.size.height)));
            }
            if (ev.type == sf::Event::MouseMoved) {
                int col = static_cast<int>((ev.mouseMove.x + gridOffset.x) / cfg.squareSize);
                int row = static_cast<int>((ev.mouseMove.y + gridOffset.y) / cfg.squareSize);
                hoveredCell = {col, row};
            }
        }

        // animazione opzionale (lasciata a 0 per lo screenshot)
        gridOffset.x = std::fmod(gridOffset.x + cfg.speed, cfg.squareSize);
        gridOffset.y = std::fmod(gridOffset.y + cfg.speed, cfg.squareSize);

        // aggiorna opacità hover
        std::unordered_map<std::string, float> targets;
        if (hoveredCell.x >= 0) {
            targets[key(hoveredCell.x, hoveredCell.y)] = cfg.hoverMaxAlpha;
        }

        // inizializza nuove celle
        for (auto &t : targets) {
            if (!cellOpacities.count(t.first)) cellOpacities[t.first] = 0.f;
        }

        // fade in/out
        for (auto it = cellOpacities.begin(); it!= cellOpacities.end();) {
            float target = targets.count(it->first)? targets[it->first] : 0.f;
            it->second += (target - it->second) * cfg.fadeSpeed;
            if (it->second < 0.005f && target == 0.f) {
                it = cellOpacities.erase(it);
            } else {
                ++it;
            }
        }

        window.clear(cfg.background);

        int cols = static_cast<int>(window.getSize().x / cfg.squareSize) + 3;
        int rows = static_cast<int>(window.getSize().y / cfg.squareSize) + 3;
        float offX = -std::fmod(gridOffset.x, cfg.squareSize);
        float offY = -std::fmod(gridOffset.y, cfg.squareSize);

        for (int c = -1; c < cols; c++) {
            for (int r = -1; r < rows; r++) {
                float x = c * cfg.squareSize + offX;
                float y = r * cfg.squareSize + offY;

                // fill hover
                std::string k = key(c, r);
                if (cellOpacities.count(k)) {
                    float a = cellOpacities[k];
                    sf::RectangleShape fill({cfg.squareSize, cfg.squareSize});
                    fill.setPosition(x, y);
                    sf::Color col = cfg.hoverFill;
                    col.a = static_cast<sf::Uint8>(a * 255);
                    fill.setFillColor(col);
                    window.draw(fill);
                }

                // bordo griglia
                sf::RectangleShape cell({cfg.squareSize, cfg.squareSize});
                cell.setPosition(x, y);
                cell.setFillColor(sf::Color::Transparent);
                cell.setOutlineThickness(1.f);
                cell.setOutlineColor(cfg.borderColor);
                window.draw(cell);
            }
        }

        window.display();
    }
    return 0;
