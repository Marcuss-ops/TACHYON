# `--cpp` Scene Loader Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Sostituire `--scene file.json --job file.json` con `--cpp scene.cpp --out video.mp4`, compilando il file utente come DLL e caricandola via `LoadLibrary`/`dlopen` per ottenere la `SceneSpec` direttamente in C++.

**Architecture:** Il CLI compila il `.cpp` utente con il compilatore di sistema (MSVC su Windows, clang/gcc su Linux), usa `LoadLibrary`/`dlopen` per caricare la DLL risultante, chiama `GetProcAddress("build_scene")` per ottenere la `SceneSpec`, poi passa tutto al renderer esistente. I parametri di output (codec, frame range, qualità) restano flags CLI. Alla fine, tutto il codice JSON parsing viene eliminato.

**Tech Stack:** C++17, MSVC `cl.exe` / `clang++`, `LoadLibrary`/`dlopen`, CMake `configure_file` per embed dei path.

```cpp
#include "tachyon/core/cpp_scene_loader.h"
#include "tachyon/core/tachyon_build_config.h"
#include <cstdlib>
#include <filesystem>
#include <format>
#include <sstream>

#ifdef _WIN32
#  define WIN32_LEAN_AND_MEAN
#  include <windows.h>
#else
#  include <dlfcn.h>
#endif

namespace tachyon {

namespace {
namespace fs = std::filesystem;

// Ritorna il path alla DLL/SO temporanea dato il sorgente.
fs::path dll_path_for(const fs::path& src, const fs::path& out_dir) {
    const fs::path base = out_dir.empty()
        ? fs::temp_directory_path()
        : out_dir;
    const std::string stem = src.stem().string() + "_scene";
#ifdef _WIN32
    return base / (stem + ".dll");
#else
    return base / ("lib" + stem + ".so");
#endif
}

// Compila src → dll. Ritorna stringa di errore vuota se ok.
std::string compile(const fs::path& src, const fs::path& dll) {
    std::string cmd;
#ifdef _WIN32
    // Trova cl.exe tramite vswhere o PATH
    cmd = std::format(
        "cl.exe /nologo /LD /std:c++17 /EHsc /MD"
        " /I\"{}\" "
        " \"{}\" "
        " /Fe\"{}\" "
        " /link \"{}\" "
        " > nul 2>&1",
        TACHYON_INCLUDE_DIR,
        src.string(),
        dll.string(),
        TACHYON_LIB_PATH
    );
#else
    cmd = std::format(
        "c++ -shared -fPIC -std=c++17"
        " -I\"{}\" "
        " \"{}\" "
        " -o \"{}\" "
        " \"{}\" "
        " 2>/dev/null",
        TACHYON_INCLUDE_DIR,
        src.string(),
        dll.string(),
        TACHYON_LIB_PATH
    );
#endif
    const int ret = std::system(cmd.c_str());
    if (ret != 0) {
        // Ricompila con output visibile per il messaggio di errore
#ifdef _WIN32
        std::string dbg = std::format(
            "cl.exe /nologo /LD /std:c++17 /EHsc /MD"
            " /I\"{}\" \"{}\" /Fe\"{}\" /link \"{}\"",
            TACHYON_INCLUDE_DIR, src.string(), dll.string(), TACHYON_LIB_PATH);
#else
        std::string dbg = std::format(
            "c++ -shared -fPIC -std=c++17 -I\"{}\" \"{}\" -o \"{}\" \"{}\"",
            TACHYON_INCLUDE_DIR, src.string(), dll.string(), TACHYON_LIB_PATH);
#endif
        return std::format("Compilation failed.\nCommand: {}", dbg);
    }
    return {};
}

// Carica la DLL e chiama build_scene(). Ritorna CppSceneLoadResult.
CppSceneLoadResult load_and_call(const fs::path& dll) {
    using BuildSceneFn = SceneSpec(*)();

#ifdef _WIN32
    HMODULE handle = LoadLibraryW(dll.wstring().c_str());
    if (!handle) {
        return {{}, std::format("LoadLibrary failed: error {}", GetLastError())};
    }
    auto fn = reinterpret_cast<BuildSceneFn>(
        GetProcAddress(handle, "build_scene"));
    if (!fn) {
        FreeLibrary(handle);
        return {{}, "build_scene() not found in compiled scene. Did you forget extern \"C\" TACHYON_API SceneSpec build_scene()?"};
    }
    SceneSpec spec = fn();
    FreeLibrary(handle);
#else
    void* handle = dlopen(dll.c_str(), RTLD_NOW | RTLD_LOCAL);
    if (!handle) {
        return {{}, std::format("dlopen failed: {}", dlerror())};
    }
    auto fn = reinterpret_cast<BuildSceneFn>(dlsym(handle, "build_scene"));
    if (!fn) {
        dlclose(handle);
        return {{}, "build_scene() not found in compiled scene."};
    }
    SceneSpec spec = fn();
    dlclose(handle);
#endif
    return {std::move(spec), {}};
}
} // namespace

CppSceneLoadResult load_cpp_scene(
    const fs::path& src_path,
    const fs::path& dll_out_dir)
{
    if (!fs::exists(src_path))
        return {{}, std::format("File not found: {}", src_path.string())};

    const fs::path dll = dll_path_for(src_path, dll_out_dir);

    if (const auto err = compile(src_path, dll); !err.empty())
        return {{}, err};

    return load_and_call(dll);
}

} // namespace tachyon
```

```cpp
// Costruisce un RenderJob dai flag CLI.
RenderJob make_job_from_options(const SceneSpec& scene, const CliOptions& opts) {
    RenderJob job;
    job.job_id = "cpp_render";
    job.composition_target = scene.compositions.front().id;

    const auto& comp = scene.compositions.front();
    const std::int64_t total_frames =
        static_cast<std::int64_t>(comp.duration * comp.frame_rate.value());

    job.frame_range = opts.frame_range_override.value_or(FrameRange{0, total_frames});

    // Se --frame N è specificato, renderizza solo quel frame
    if (opts.single_frame.has_value()) {
        job.frame_range = {*opts.single_frame, *opts.single_frame + 1};
    }

    const std::filesystem::path out = opts.output_override.empty()
        ? std::filesystem::path("output.mp4")
        : opts.output_override;

    job.output.destination.path    = out.string();
    job.output.destination.overwrite = true;

    // Formato dall'estensione
    const std::string ext = out.extension().string();
    if (ext == ".png" || ext == ".PNG") {
        job.output.profile.container       = "png";
        job.output.profile.video.codec     = "png";
        job.output.profile.video.pixel_format = "rgb8";
    } else {
        job.output.profile.container          = "mp4";
        job.output.profile.video.codec        = "libx264";
        job.output.profile.video.pixel_format = "yuv420p";
        // Qualità
        if      (opts.quality == "low")    job.output.profile.video.crf = 28.0;
        else if (opts.quality == "medium") job.output.profile.video.crf = 23.0;
        else                               job.output.profile.video.crf = 18.0; // high
        job.output.profile.video.rate_control_mode = "crf";
    }
    job.output.profile.audio.mode = "none";
    return job;
}

} // namespace

bool run_render_command(const CliOptions& options, std::ostream& out, std::ostream& err) {
    SceneSpec scene;

    if (!options.cpp_path.empty()) {
        const auto result = load_cpp_scene(options.cpp_path);
        if (!result.ok()) {
            err << "Error loading scene: " << result.error << "\n";
            return false;
        }
        scene = *result.scene;
    } else if (options.preset_id.has_value()) {
        if (!build_scene_from_preset(*options.preset_id, scene, err)) return false;
    } else {
        err << "Either --cpp or --preset must be provided.\n";
        return false;
    }

    const RenderJob job = make_job_from_options(scene, options);

    NativeRenderer::Options native_options;
    native_options.worker_count        = options.worker_count;
    native_options.memory_budget_bytes = options.memory_budget_bytes;
    native_options.verbose             = !options.json_output;

    const RenderSessionResult session_result = NativeRenderer::render(scene, job, native_options);

    if (!session_result.output_error.empty()) {
        err << "Render error: " << session_result.output_error << "\n";
    }
    print_diagnostics(session_result.diagnostics, err);

    if (options.json_output) {
        out << "{\"status\": \"ok\", \"frames\": " << session_result.frames.size() << "}\n";
    } else {
        RasterizedFrame2D first_frame;
        if (!session_result.frames.empty())
            first_frame.backend_name = "cpu-frame-executor";
        const auto plan_result = build_render_plan(scene, job);
        if (plan_result.value.has_value()) {
            const auto exec = build_render_execution_plan(*plan_result.value, 0);
            if (exec.value.has_value())
                print_execution_plan(*exec.value, first_frame, session_result, out);
        }
    }
    return session_result.output_error.empty();
}

bool run_preview_command(const CliOptions& options, std::ostream& out, std::ostream& err) {
    SceneSpec scene;

    if (!options.cpp_path.empty()) {
        const auto result = load_cpp_scene(options.cpp_path);
        if (!result.ok()) { err << result.error << "\n"; return false; }
        scene = *result.scene;
    } else if (options.preset_id.has_value()) {
        if (!build_scene_from_preset(*options.preset_id, scene, err)) return false;
    } else {
        err << "Either --cpp or --preset is required for preview.\n";
        return false;
    }

    const std::string comp_id = scene.compositions.front().id;
    const std::int64_t frame  = options.single_frame.value_or(0);
    const std::filesystem::path output =
        options.preview_output.empty() ? "preview.png" : options.preview_output;

    const bool ok = NativeRenderer::render_still(scene, comp_id, frame, output);
    if (!ok) err << "Preview render failed.\n";
    return ok;
}
```

```cpp
extern "C" TACHYON_API SceneSpec build_scene() {
    // Background: noise silver
    LayerSpec bg = procedural_bg::brushed_metal(1280, 720);

    // Titolo principale
    TextParams title;
    title.text           = "10 MINUTES";
    title.font_size      = 130;
    title.color          = ColorSpec{18, 18, 20, 255};
    title.animation      = "tracking_reveal";
    title.reveal_duration = 0.6;
    title.x              = 0.0f;
    title.y              = 260.0f;
    title.w              = 1280.0f;
    title.h              = 200.0f;
    title.in_point       = 0.0;
    title.out_point      = 3.0;

    // Sottotitolo
    TextParams sub;
    sub.text           = "TIME OF SURVIVE";
    sub.font_size      = 42;
    sub.color          = ColorSpec{90, 90, 95, 255};
    sub.animation      = "fade_up";
    sub.reveal_duration = 0.5;
    sub.x              = 0.0f;
    sub.y              = 390.0f;
    sub.w              = 1280.0f;
    sub.h              = 100.0f;
    sub.in_point       = 0.2;
    sub.out_point      = 3.0;

    return scene::Composition("main")
        .size(1280, 720)
        .duration(3.0)
        .fps(30)
        .layer("bg",       [&](scene::LayerBuilder& l){ l = scene::LayerBuilder(bg); })
        .layer("title",    [&](scene::LayerBuilder& l){ l = scene::LayerBuilder(build_text(title)); })
        .layer("subtitle", [&](scene::LayerBuilder& l){ l = scene::LayerBuilder(build_text(sub)); })
        .build_scene();
}
```
